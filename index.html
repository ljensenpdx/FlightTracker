function renderFlightRow(flight, flightNum) { const board = document.getElementById('flightBoard'); const row = document.createElement('div'); // Check if arrival data exists to prevent errors on cancelled/unscheduled flights if (!flight.arrival || !flight.arrival.scheduledTimeLocal) return; // THE FIX: Replace the space with a 'T' so JavaScript can read the date properly const safeScheduled = flight.arrival.scheduledTimeLocal.replace(' ', 'T'); const scheduledTime = new Date(safeScheduled); let estimatedTime = scheduledTime; if (flight.arrival.actualTimeLocal) { const safeEstimated = flight.arrival.actualTimeLocal.replace(' ', 'T'); estimatedTime = new Date(safeEstimated); } const formatTime = (date) => date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', hour12: false }); // Calculate operational delay thresholds const timeDiffMinutes = (estimatedTime - scheduledTime) / (1000 * 60); let statusClass = "text-white"; let remarks = "ON TIME"; if (timeDiffMinutes < 0) { statusClass = "text-green"; remarks = "EARLY"; } else if (timeDiffMinutes > 0 && timeDiffMinutes <= 30) { statusClass = "text-yellow"; remarks = "DELAYED"; } else if (timeDiffMinutes > 30) { statusClass = "text-red"; remarks = "LATE"; } row.className = `flight-row ${statusClass}`; row.innerHTML = `
${flightNum}
${flight.departure.airport.iata || '---'}
${formatTime(scheduledTime)}
${formatTime(estimatedTime)}
${remarks}
`; board.appendChild(row); }
